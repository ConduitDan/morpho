35 static inline int morpho_comparevalue (value a, value b) {
36  if (!morpho_ofsametype(a, b)) return NOTEQUAL;
37 
38  if (MORPHO_ISFLOAT(a)) {
39  double x = MORPHO_GETFLOATVALUE(b) - MORPHO_GETFLOATVALUE(a);
40  if (x>DBL_EPSILON) return BIGGER; /* Fast way out for clear cut cases */
41  if (x<-DBL_EPSILON) return SMALLER;
42  /* Assumes absolute tolerance is the same as relative tolerance. */
43  if (fabs(x)<=DBL_EPSILON*fmax(1.0, fmax(MORPHO_GETFLOATVALUE(a), MORPHO_GETFLOATVALUE(b)))) return EQUAL;
44  return (x>0 ? BIGGER : SMALLER);
45  } else {
46  switch (MORPHO_GETTYPE(a)) {
47  case VALUE_NIL:
48  return EQUAL;
49  case VALUE_INTEGER:
50  return (MORPHO_GETINTEGERVALUE(b) - MORPHO_GETINTEGERVALUE(a));
51  case VALUE_BOOL:
52  return (MORPHO_GETBOOLVALUE(b) != MORPHO_GETBOOLVALUE(a));
53  case VALUE_OBJECT:
54  {
55  if (MORPHO_GETOBJECTTYPE(a)!=MORPHO_GETOBJECTTYPE(b)) {
56  return 1; /* Objects of different type are always different */
